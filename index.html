# 멋쟁이사자처럼 순천대학교

class LikeLion:
    def __init__(self):
        self.university = "순천대학교"
        self.mission = "아이디어를 실현하는 IT 창업 동아리"
        self.members = ["기획자", "프론트엔드", "백엔드", "디자이너"]
        self.activities = {
            "정기_세션": "주 1회 개발 교육",
            "팀_프로젝트": "중간/최종 서비스 개발",
            "스터디": "자율 학습 그룹",
            "메인_프로젝트": "전국 연합 아이디어톤과 해커톤"
        }

    def grow_together(self):
        skills = ["코딩", "협업", "문제해결"]
        return f"{skills} 역량 향상"

    def deploy_service(self):
        return "기획 → 개발 → 배포까지 완성"

# 전국 최대 규모의 IT 창업 동아리
likelion = LikeLion()
print(likelion.grow_together())
print(likelion.deploy_service())
